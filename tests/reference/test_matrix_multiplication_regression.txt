def matmul(C: BufferizedNDArray, A: BufferizedNDArray, B: BufferizedNDArray) -> BufferizedNDArray:
    A_: BufferizedNDArray = A
    A__stride_0: finchlite.int64 = A_.strides.element_0
    A__stride_1: finchlite.int64 = A_.strides.element_1
    A__buf: np_buf_t(float64) = A_.val
    A__buf_slot: np_buf_t(float64) = unpack(A__buf)
    B_: BufferizedNDArray = B
    B__stride_0: finchlite.int64 = B_.strides.element_0
    B__stride_1: finchlite.int64 = B_.strides.element_1
    B__buf: np_buf_t(float64) = B_.val
    B__buf_slot: np_buf_t(float64) = unpack(B__buf)
    C_: BufferizedNDArray = C
    C__stride_0: finchlite.int64 = C_.strides.element_0
    C__stride_1: finchlite.int64 = C_.strides.element_1
    C__buf: np_buf_t(float64) = C_.val
    C__buf_slot: np_buf_t(float64) = unpack(C__buf)
    m: finchlite.int64 = A_.shape.element_0
    n: finchlite.int64 = B_.shape.element_1
    p: finchlite.int64 = A_.shape.element_1
    for i in range(0, length(slot(C__buf_slot, np_buf_t(float64)))):
        store(slot(C__buf_slot, np_buf_t(float64)), i, 0.0)
    for i in range(0, m):
        i__pos: finchlite.int64 = add(0, mul(A__stride_0, i))
        i__pos_2: finchlite.int64 = add(0, mul(C__stride_0, i))
        for k in range(0, p):
            k__pos: finchlite.int64 = add(i__pos, mul(A__stride_1, k))
            k__pos_2: finchlite.int64 = add(0, mul(B__stride_0, k))
            for j in range(0, n):
                j__pos: finchlite.int64 = add(k__pos_2, mul(B__stride_1, j))
                j__pos_2: finchlite.int64 = add(i__pos_2, mul(C__stride_1, j))
                a_ik: finchlite.float64 = load(slot(A__buf_slot, np_buf_t(float64)), k__pos)
                b_kj: finchlite.float64 = load(slot(B__buf_slot, np_buf_t(float64)), j__pos)
                c_ij: finchlite.float64 = mul(a_ik, b_kj)
                store(slot(C__buf_slot, np_buf_t(float64)), j__pos_2, add(load(slot(C__buf_slot, np_buf_t(float64)), j__pos_2), c_ij))
    repack(C__buf_slot)
    matmul_return: BufferizedNDArray = C
    return matmul_return
